**1. Add `write_pomodoro_command` to `control_panel/config_io.py`:**

Add a new function after `atomic_write_config`, using the exact same atomic temp+replace pattern:
```python
def write_pomodoro_command(config_dir: str, command: str) -> None:
    """Write a Pomodoro control command atomically.

    Writes {"cmd": command} to pomodoro_command.json in config_dir.
    The host's QFileSystemWatcher picks up the change, forwards it
    as a ControlSignal, and deletes the file.
    """
    cmd_path = os.path.join(os.path.abspath(config_dir), "pomodoro_command.json")
    dir_path = os.path.dirname(cmd_path)
    fd, tmp_path = tempfile.mkstemp(dir=dir_path, suffix=".tmp")
    try:
        with os.fdopen(fd, "w", encoding="utf-8") as f:
            json.dump({"cmd": command}, f)
        os.replace(tmp_path, cmd_path)
    except Exception:
        try:
            os.unlink(tmp_path)
        except OSError:
            pass
        raise
```

```python
import json as _json  # avoid shadowing; or just use json (already imported indirectly via config_loader)

# --- Command-file watcher for Pomodoro controls (POMO-04) ---
config_dir = os.path.dirname(os.path.abspath("config.json"))
cmd_path = os.path.join(config_dir, "pomodoro_command.json")

cmd_watcher = QFileSystemWatcher()
# Only add path if file exists; fileChanged will re-add after each atomic replace
if os.path.exists(cmd_path):
    cmd_watcher.addPath(cmd_path)

def _on_cmd_file_changed(path: str):
    """Read command file, forward ControlSignal to Pomodoro widget, delete file."""
    cmd_watcher.addPath(cmd_path)  # re-add after atomic replace (same pattern as config watcher)
    if not os.path.exists(cmd_path):
        return
    try:
        with open(cmd_path, encoding="utf-8") as f:
            data = _json.load(f)
        command = data.get("cmd")
        if command in ("start", "pause", "reset"):
            pm.send_control_signal("pomodoro", command)
            print(f"[CommandWatcher] forwarded '{command}' to pomodoro", flush=True)
        os.unlink(cmd_path)
    except (OSError, _json.JSONDecodeError) as exc:
        print(f"[CommandWatcher] error reading command file: {exc}", flush=True)

cmd_watcher.fileChanged.connect(_on_cmd_file_changed)
window._cmd_watcher = cmd_watcher  # prevent GC
```

Add these imports to the top of host/main.py:

**Important:** The command file watcher must also handle the case where the file doesn't exist yet at startup. QFileSystemWatcher.addPath() returns False for non-existent paths, which is fine — the watcher will be effective once _on_cmd_file_changed re-adds the path after the first write. To ensure the watcher can pick up the first command, we need a different approach: watch the directory instead, or create the file initially.

**Better approach:** Watch the *directory* for changes and check for the command file:
```python
# Watch the directory so we catch file creation (not just modification)
cmd_watcher.addPath(config_dir)

def _on_cmd_dir_changed(path: str):
    """Check if pomodoro_command.json appeared or changed."""
    if not os.path.exists(cmd_path):
        return
    try:
        with open(cmd_path, encoding="utf-8") as f:
            data = json.load(f)
        command = data.get("cmd")
        if command in ("start", "pause", "reset"):
            pm.send_control_signal("pomodoro", command)
            print(f"[CommandWatcher] forwarded '{command}' to pomodoro", flush=True)
        os.unlink(cmd_path)
    except (OSError, json.JSONDecodeError) as exc:
        print(f"[CommandWatcher] error: {exc}", flush=True)

cmd_watcher.directoryChanged.connect(_on_cmd_dir_changed)
window._cmd_watcher = cmd_watcher
```

Use `directoryChanged` signal (watches directory for new/removed files) rather than `fileChanged` (requires file to exist). This correctly handles the first command write when the file doesn't exist yet.
